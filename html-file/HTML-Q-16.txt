<?php
// Function to add two numbers
function addNumbers($a, $b) {
  return $a + $b;
}

// Check if form is submitted
if (isset($_POST['submit'])) {
  $num1 = (int)$_POST['num1'];
  $num2 = (int)$_POST['num2'];
  $sum = addNumbers($num1, $num2);

  echo "<h3>Result: $num1 + $num2 = $sum</h3>";
}
